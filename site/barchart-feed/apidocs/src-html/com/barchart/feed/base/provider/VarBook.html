008    package com.barchart.feed.base.provider;
009    
010    import static com.barchart.feed.base.book.enums.MarketBookSide.ASK;
011    import static com.barchart.feed.base.book.enums.MarketBookSide.BID;
012    import static com.barchart.feed.base.provider.MarketConst.NULL_BOOK_ENTRY;
013    
014    import com.barchart.feed.base.book.api.MarketBook;
015    import com.barchart.feed.base.book.api.MarketBookEntry;
016    import com.barchart.feed.base.book.api.MarketDoBook;
017    import com.barchart.feed.base.book.api.MarketDoBookEntry;
018    import com.barchart.feed.base.book.enums.MarketBookSide;
019    import com.barchart.feed.base.book.enums.MarketBookType;
020    import com.barchart.feed.base.book.enums.UniBookResult;
021    import com.barchart.util.anno.Mutable;
022    import com.barchart.util.anno.ThreadSafe;
023    import com.barchart.util.values.api.PriceValue;
024    import com.barchart.util.values.api.SizeValue;
025    import com.barchart.util.values.api.TimeValue;
026    import com.barchart.util.values.provider.ValueBuilder;
027    
028    @Mutable
029    @ThreadSafe(rule = "use in runSafe() only")
030    public final class VarBook extends UniBook<MarketBook> implements MarketDoBook {
031    
032            private long millisUTC;
033    
034            public VarBook(final MarketBookType type, final SizeValue size,
035                            final PriceValue step) {
036                    super(type, size, step);
037            }
038    
039            @Override
040            public final UniBookResult setEntry(final MarketDoBookEntry entry) {
041                    return make(entry);
042            }
043    
044            @Override
045            public final MarketBookEntry[] entries(final MarketBookSide side) {
046                    return entriesFor(side);
047            }
048    
049            @Override
050            public final DefBook freeze() {
051                    return new DefBook(time(), entries(BID), entries(ASK), null, null);
052            }
053    
054            @Override
055            public TimeValue time() {
056                    return ValueBuilder.newTime(millisUTC);
057            }
058    
059            @Override
060            public final void setTime(final TimeValue time) {
061                    millisUTC = time.asMillisUTC();
062            }
063    
064            @Override
065            public final boolean isFrozen() {
066                    return false;
067            }
068    
069            @Override
070            public final MarketBookEntry last() {
071                    final DefBookEntry entry = lastEntry();
072                    return entry == null ? NULL_BOOK_ENTRY : entry;
073            }
074    
075            @Override
076            public final MarketBookEntry top(final MarketBookSide side) {
077                    final MarketDoBookEntry entry = topFor(side);
078                    return entry == null ? NULL_BOOK_ENTRY : entry;
079            }
080    
081            /* #################################### */
082    
083            @Override
084            public PriceValue priceGap() {
085                    throw new UnsupportedOperationException("UNUSED");
086            }
087    
088            @Override
089            public final SizeValue[] sizes(final MarketBookSide side) {
090                    throw new UnsupportedOperationException("UNUSED");
091            }
092    
093            @Override
094            public PriceValue priceTop(final MarketBookSide side) {
095                    throw new UnsupportedOperationException("UNUSED");
096            }
097    
098            @Override
099            public SizeValue sizeTop(final MarketBookSide side) {
100                    throw new UnsupportedOperationException("UNUSED");
101            }
102    
103            @Override
104            public UniBookResult setSnapshot(final MarketDoBookEntry[] entries) {
105                    throw new UnsupportedOperationException("UNUSED");
106            }
107    
108    }
