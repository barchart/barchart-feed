008    package com.barchart.feed.base.cuvol.map.provider;
009    
010    import java.util.Map;
011    
012    import com.barchart.util.collections.PriceArrayMap;
013    import com.barchart.util.values.api.PriceValue;
014    import com.barchart.util.values.api.SizeValue;
015    import com.barchart.util.values.provider.ValueConst;
016    import com.barchart.util.values.util.ValueUtil;
017    
018    public class DefCuvolMap extends NulCuvolMap {
019    
020            public static double EPSILON = 0.0000000001;
021            
022            private final Map<PriceValue, SizeValue> offGridMap;
023            
024            private final SizeValue[] entries;
025            private final PriceArrayMap<SizeValue> map;
026    
027            private final PriceValue priceFirst;
028            private final PriceValue priceStep;
029            
030            DefCuvolMap(final Map<PriceValue, SizeValue> offGridMap, PriceArrayMap<SizeValue> map, 
031                            final PriceValue priceFirst, final PriceValue priceStep) {
032    
033                    assert offGridMap != null;
034                    assert priceFirst != null;
035                    assert priceStep != null;
036                    assert priceStep.mantissa() != 0;
037    
038                    this.offGridMap = offGridMap;
039                    
040                    this.priceFirst = priceFirst;
041                    this.priceStep = priceStep;
042    
043                    this.map = map;
044                    
045                    entries = entries();
046    
047                    for (int k = 0; k < entries.length; k++) {
048                            SizeValue volume = entries[k];
049                            if (volume == null) {
050                                    volume = ValueConst.NULL_SIZE;
051                            }
052                            entries[k] = volume.freeze();
053                    }
054                    
055            }
056            
057            @Override
058            public SizeValue getCuvol(final PriceValue price) {
059                    
060                    SizeValue size = ValueConst.NULL_SIZE;
061                    if(onGrid(price)) {
062                            size = map.get(price);
063                    } else if(offGridMap != null) {
064                            size = offGridMap.get(price);
065                    }
066                    
067                    if(size == null) {
068                            return ValueConst.NULL_SIZE;
069                    } else {
070                            return size;
071                    }
072            }
073            
074            @Override
075            public PriceValue priceFirst() {
076                    return priceFirst;
077            }
078    
079            @Override
080            public PriceValue priceStep() {
081                    return priceStep;
082            }
083    
084            @Override
085            public SizeValue[] entries() {
086                    return entries;
087            }
088            
089            private boolean onGrid(final PriceValue price) {
090                    
091                    //Do without double
092                    final double result = ValueUtil.asDouble(price) / ValueUtil.asDouble(priceStep);
093                    
094                    final long resLong = Math.round(result);
095                    
096                    if(Math.abs(result - resLong) < EPSILON) {
097                            return true;
098                    }
099                    
100                    return false;
101                    
102            }
103            
104    }
