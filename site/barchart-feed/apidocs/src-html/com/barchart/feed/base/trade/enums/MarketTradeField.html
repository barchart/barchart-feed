008    package com.barchart.feed.base.trade.enums;
009    
010    import static com.barchart.feed.base.trade.enums.MarketTradeSequencing.NULL_TRADE_SEQUENCE;
011    import static com.barchart.feed.base.trade.enums.MarketTradeSession.NULL_TRADE_SESSION;
012    import static com.barchart.feed.base.trade.enums.MarketTradeType.NULL_TRADE_TYPE;
013    import static com.barchart.util.values.provider.ValueConst.NULL_PRICE;
014    import static com.barchart.util.values.provider.ValueConst.NULL_SIZE;
015    import static com.barchart.util.values.provider.ValueConst.NULL_TIME;
016    
017    import com.barchart.util.anno.NotMutable;
018    import com.barchart.util.collections.BitSetEnum;
019    import com.barchart.util.enums.DictEnum;
020    import com.barchart.util.enums.ParaEnumBase;
021    import com.barchart.util.values.api.PriceValue;
022    import com.barchart.util.values.api.SizeValue;
023    import com.barchart.util.values.api.TimeValue;
024    import com.barchart.util.values.api.Value;
025    
026    @NotMutable
027    public final class MarketTradeField<V extends Value<V>> extends
028                    ParaEnumBase<V, MarketTradeField<V>> implements
029                    BitSetEnum<MarketTradeField<?>> {
030    
031            // ##################################
032    
033            /** where trade originated */
034            public static final MarketTradeField<MarketTradeType> TYPE = NEW(NULL_TRADE_TYPE);
035    
036            /** where trade originated */
037            public static final MarketTradeField<MarketTradeSession> SESSION = NEW(NULL_TRADE_SESSION);
038    
039            /** where trade originated */
040            public static final MarketTradeField<MarketTradeSequencing> SEQUENCING = NEW(NULL_TRADE_SEQUENCE);
041    
042            /** last trade price */
043            public static final MarketTradeField<PriceValue> PRICE = NEW(NULL_PRICE);
044    
045            /** last trade size */
046            public static final MarketTradeField<SizeValue> SIZE = NEW(NULL_SIZE);
047    
048            /** last trade time */
049            public static final MarketTradeField<TimeValue> TRADE_TIME = NEW(NULL_TIME);
050    
051            /** last trade day/session */
052            public static final MarketTradeField<TimeValue> TRADE_DATE = NEW(NULL_TIME);
053    
054            // ##################################
055    
056            public static int size() {
057                    return values().length;
058            }
059    
060            public static MarketTradeField<?>[] values() {
061                    return DictEnum.valuesForType(MarketTradeField.class);
062            }
063    
064            private final long mask;
065    
066            @Override
067            public long mask() {
068                    return mask;
069            }
070    
071            private MarketTradeField() {
072                    super();
073                    mask = 0;
074            }
075    
076            private MarketTradeField(final V value) {
077                    super("", value);
078                    mask = ONE << ordinal();
079            }
080    
081            private static <X extends Value<X>> MarketTradeField<X> NEW(
082                            final X defaultValue) {
083                    return new MarketTradeField<X>(defaultValue);
084            }
085    
086    }
