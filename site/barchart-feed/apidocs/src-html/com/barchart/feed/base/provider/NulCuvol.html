008    package com.barchart.feed.base.provider;
009    
010    import com.barchart.feed.base.cuvol.api.MarketCuvol;
011    import com.barchart.util.anno.NotMutable;
012    import com.barchart.util.values.api.PriceValue;
013    import com.barchart.util.values.api.SizeValue;
014    import com.barchart.util.values.provider.ValueConst;
015    import com.barchart.util.values.provider.ValueFreezer;
016    
017    @NotMutable
018    public class NulCuvol extends ValueFreezer<MarketCuvol> implements MarketCuvol {
019    
020            @Override
021            public PriceValue priceFirst() {
022                    return ValueConst.NULL_PRICE;
023            }
024    
025            @Override
026            public PriceValue priceStep() {
027                    return ValueConst.NULL_PRICE;
028            }
029    
030            @Override
031            public SizeValue[] entries() {
032                    return ValueConst.NULL_SIZE_ARRAY;
033            }
034    
035            @Override
036            public final String toString() {
037    
038                    final StringBuilder text = new StringBuilder(512);
039    
040                    text.append("Cumulative Volume");
041                    text.append("\n");
042                    text.append("Price Step :  ");
043                    text.append(priceStep());
044                    text.append("\n");
045                    text.append("Price First : ");
046                    text.append(priceFirst());
047                    text.append("\n");
048    
049                    for (final SizeValue entry : entries()) {
050                            text.append(entry);
051                            text.append("\n");
052                    }
053    
054                    return text.toString();
055    
056            }
057    
058            @Override
059            public final boolean isNull() {
060                    return this == MarketConst.NULL_CUVOL;
061            }
062    
063    }
