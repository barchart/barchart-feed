008    package com.barchart.feed.base.market.api;
009    
010    import com.barchart.feed.base.instrument.values.MarketInstrument;
011    import com.barchart.feed.base.market.enums.MarketField;
012    import com.barchart.util.anno.NotMutable;
013    import com.barchart.util.anno.NotYetImplemented;
014    import com.barchart.util.anno.ThreadSafe;
015    
016    /** market provider interface */
017    @ThreadSafe
018    public interface MarketMakerProvider<Message extends MarketMessage> extends
019                    MarketMaker {
020    
021            /**
022             * remove all markets and all takers
023             */
024            void clearAll();
025    
026            //
027    
028            /**
029             * note: fires events from invocation thread
030             */
031            void make(@NotMutable Message message);
032    
033            //
034    
035            @NotYetImplemented
036            void copyTo(MarketMakerProvider<Message> maker, MarketField<?>... fields);
037    
038            //
039    
040            /**
041             * add listener that will receive notifications when any taker is registered
042             * or un-registered
043             */
044            void add(MarketRegListener listener);
045    
046            /** */
047            void remove(MarketRegListener listener);
048    
049            /**
050             * notify all registered market registration listeners for all registered
051             * markets
052             */
053            void notifyRegListeners();
054    
055            //
056    
057            /** TODO */
058            void appendMarketProvider(MarketFactory marketFactory);
059    
060            /** register market w/o takers */
061            boolean register(MarketInstrument instrument);
062    
063            /** unregister market w/o takers */
064            boolean unregister(MarketInstrument instrument);
065    
066    }
