060    public class MockMarket extends VarMarket {
061    
062            @Override
063            public void setInstrument(final MarketInstrument symbol) {
064    
065                    final MarketInstrument oldInst = get(INSTRUMENT);
066    
067                    if (InstrumentConst.NULL_INSTRUMENT.equals(oldInst)) {
068                            set(INSTRUMENT, symbol);
069                    } else {
070                            throw new IllegalStateException("symbol can be set only once");
071                    }
072            }
073    
074            @Override
075            public void setBookUpdate(final MarketDoBookEntry entry,
076                            final TimeValue time) {
077    
078                    assert entry != null && time != null;
079    
080                    final MarketDoBook book = (MarketDoBook) get(BOOK);
081    
082                    final UniBookResult result = book.setEntry(entry);
083    
084                    switch (result) {
085                    case TOP:
086                            eventAdd(NEW_BOOK_TOP);
087                            // continue
088                    case NORMAL:
089                            eventAdd(NEW_BOOK_UPDATE);
090                            break;
091                    default:
092                            eventAdd(NEW_BOOK_ERROR);
093                            return;
094                    }
095    
096                    book.setTime(time);
097                    updateMarket(time);
098    
099            }
100    
101            @Override
102            public void setBookSnapshot(final MarketDoBookEntry[] entries,
103                            final TimeValue time) {
104                    assert entries != null;
105                    assert time != null;
106    
107                    final MarketDoBook book = (MarketDoBook) get(BOOK);
108    
109                    book.clear();
110    
111                    for (final MarketDoBookEntry entry : entries) {
112    
113                            if (entry == null) {
114                                    continue;
115                            }
116    
117                            final UniBookResult result = book.setEntry(entry);
118    
119                            switch (result) {
120                            case TOP:
121                            case NORMAL:
122                                    break;
123                            default:
124                                    eventAdd(NEW_BOOK_ERROR);
125                                    break;
126                            }
127    
128                    }
129    
130                    eventAdd(NEW_BOOK_SNAPSHOT);
131    
132                    book.setTime(time);
133                    updateMarket(time);
134    
135            }
136    
137            @Override
138            public void setCuvolUpdate(final MarketDoCuvolEntry entry,
139                            final TimeValue time) {
140    
141                    assert entry != null && time != null;
142    
143                    makeCuvol(entry.price(), entry.size());
144    
145                    updateMarket(time);
146    
147            }
148    
149            @Override
150            public void setCuvolSnapshot(final MarketDoCuvolEntry[] entries,
151                            final TimeValue time) {
152    
153                    assert entries != null && time != null;
154    
155                    final MarketDoCuvol cuvol = loadCuvol();
156    
157                    cuvol.clear();
158    
159                    for (final MarketDoCuvolEntry entry : entries) {
160                            cuvol.add(entry.price(), entry.size());
161                    }
162    
163                    eventAdd(NEW_CUVOL_SNAPSHOT);
164    
165                    updateMarket(time);
166    
167            }
168    
169            @Override
170            public void setBar(final MarketBarType type, final MarketDoBar bar) {
171    
172                    assert type != null;
173                    assert bar != null;
174    
175                    set(type.field, bar);
176    
177                    eventAdd(type.event);
178    
179                    updateMarket(bar.get(BAR_TIME));
180    
181            }
182    
183            @Override
184            public void setTrade(final MarketTradeType type,
185                            final MarketTradeSession session,
186                            final MarketTradeSequencing sequencing, final PriceValue price,
187                            final SizeValue size, final TimeValue time, final TimeValue date) {
188    
189                    assert type != null;
190                    assert price != null;
191                    assert size != null;
192                    assert time != null;
193                    assert date != null;
194    
195                    final MarketDoTrade trade = loadTrade();
196    
197                    trade.set(TYPE, type);
198                    trade.set(PRICE, price);
199                    trade.set(SIZE, size);
200                    trade.set(TRADE_TIME, time);
201                    trade.set(MarketTradeField.TRADE_DATE, date);
202    
203                    applyTradeToBar(CURRENT, price, size, time, date);
204    
205                    makeCuvol(price, size);
206    
207                    updateMarket(time);
208    
209            }
210    
211            private final void updateMarket(final TimeValue time) {
212    
213                    set(MARKET_TIME, time);
214    
215                    eventAdd(MARKET_UPDATED);
216    
217            }
218    
219            private final void makeCuvol(final PriceValue price, final SizeValue size) {
220    
221                    final MarketDoCuvol cuvol = loadCuvol();
222    
223                    cuvol.add(price, size);
224    
225                    eventAdd(NEW_CUVOL_UPDATE);
226    
227            }
228    
229            private final void applyTradeToBar(final MarketBarType type,
230                            final PriceValue price, final SizeValue size, final TimeValue time,
231                            final TimeValue date) {
232    
233                    final MarketDoBar bar = loadBar(type.field);
234    
235                    eventAdd(type.event);
236    
237                    final SizeValue volumeOld = bar.get(VOLUME);
238                    final SizeValue volumeNew = volumeOld.add(size);
239                    bar.set(VOLUME, volumeNew);
240                    eventAdd(NEW_VOLUME);
241    
242                    bar.set(CLOSE, price);
243                    if (type == CURRENT) {
244                            // events only for combo
245                            eventAdd(NEW_CLOSE);
246                    }
247    
248                    // ### time
249    
250                    bar.set(BAR_TIME, time);
251                    bar.set(TRADE_DATE, date);
252    
253            }
254    
255            @Override
256            public void setState(final MarketStateEntry entry, final boolean isOn) {
257    
258                    assert entry != null;
259    
260                    final MarketState state = loadState();
261    
262                    if (isOn) {
263                            state.add(entry);
264                    } else {
265                            state.remove(entry);
266                    }
267    
268            }
