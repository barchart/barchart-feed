8   package com.barchart.feed.base.provider;

10  import com.barchart.feed.base.bar.enums.MarketBarField;
11  import com.barchart.util.anno.NotMutable;
12  import com.barchart.util.values.api.Value;
13  
14  @NotMutable
15  class DefBar extends NulBar {
16  
17  	protected final static int ARRAY_SIZE = MarketBarField.size();
18  
19  	protected final Value<?>[] valueArray;
20  
21  	DefBar() {
22  		valueArray = new Value<?>[ARRAY_SIZE];
23  	}
24  
25  	DefBar(final Value<?>[] valueArray) {
26  		assert valueArray != null;
27  		assert valueArray.length == ARRAY_SIZE;
28  		this.valueArray = valueArray;
29  	}
30  
31  	@SuppressWarnings("unchecked")
32  	@Override
33  	public <V extends Value<V>> V get(final MarketBarField<V> field) {
34  
35  		assert field != null;
36  
37  		final V value = (V) valueArray[field.ordinal()];
38  
39  		if (value == null) {
40  			return field.value();
41  		} else {
42  			return value;
43  		}
44  
45  	}
46  
47  }
